97  template<class T>
98  static std::string uriModifier(T &cls, std::string uri_to_modify, eUriType type) {
99  // When uri_to_modify equals model or local uri, we throw.
100  if (
101  uri_to_modify == cls.getLocalUri() ||
102  uri_to_modify == cls.getLocalUri() + "#" ||
103  uri_to_modify == cls.getModelUri() ||
104  uri_to_modify == cls.getModelUri() + "#") {
105  throw std::logic_error("std::string uriModifier: Cannot modify input string: \"" + uri_to_modify + "\"");
106  }
107  // When we already have a uri that is not local or model uri, we just return
108  if (OmexMetaUtils::startsWith(uri_to_modify, "http")) {
109  return uri_to_modify;
110  }
111  switch (type) {
112  case NONE: {
113  return uri_to_modify;
114  }
115  case LOCAL_URI: {
116  if (cls.getLocalUri().empty()) {
117  throw std::logic_error("UriHandler::uriModifier(): Cannot create local uri for string \"" + uri_to_modify + "\"as local uri is empty");
118  }
119  return OmexMetaUtils::concatMetaIdAndUri(uri_to_modify, cls.getLocalUri());
120  }
121  case MODEL_URI: {
122  if (cls.getModelUri().empty())
123  throw std::logic_error("UriHandler::uriModifier(): Cannot create model uri for string \"" + uri_to_modify + "\"as local uri is empty");
124  return OmexMetaUtils::concatMetaIdAndUri(uri_to_modify, cls.getModelUri());
125  }
126  case IDENTIFIERS_URI: {
127  return "https://identifiers.org/" + uri_to_modify;
128  }
129  default: {
130  throw std::invalid_argument("std::string uriModifier: Unrecognized eUriType");
131  }
132  }
133  }
