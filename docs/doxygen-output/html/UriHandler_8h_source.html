81  template<class T>
82  static std::string uriModifier(T &cls, std::string uri_to_modify, eUriType type) {
83  // When uri_to_modify equals model or local uri, we throw.
84  if (
85  uri_to_modify == cls.getLocalUri() ||
86  uri_to_modify == cls.getLocalUri() + "#" ||
87  uri_to_modify == cls.getModelUri() ||
88  uri_to_modify == cls.getModelUri() + "#") {
89  throw std::logic_error("std::string uriModifier: Cannot modify input string: \"" + uri_to_modify + "\"");
90  }
91  // When we already have a uri that is not local or model uri, we just return
92  if (OmexMetaUtils::startsWith(uri_to_modify, "http")) {
93  return uri_to_modify;
94  }
95  switch (type) {
96  case NONE: {
97  return uri_to_modify;
98  }
99  case LOCAL_URI: {
100  if (cls.getLocalUri().empty()) {
101  throw std::logic_error("UriHandler::uriModifier(): Cannot create local uri for string \"" + uri_to_modify + "\"as local uri is empty");
102  }
103  return OmexMetaUtils::concatMetaIdAndUri(uri_to_modify, cls.getLocalUri());
104  }
105  case MODEL_URI: {
106  if (cls.getModelUri().empty())
107  throw std::logic_error("UriHandler::uriModifier(): Cannot create model uri for string \"" + uri_to_modify + "\"as local uri is empty");
108  return OmexMetaUtils::concatMetaIdAndUri(uri_to_modify, cls.getModelUri());
109  }
110  case IDENTIFIERS_URI: {
111  return "https://identifiers.org/" + uri_to_modify;
112  }
113  default: {
114  throw std::invalid_argument("std::string uriModifier: Unrecognized eUriType");
115  }
116  }
117  }
