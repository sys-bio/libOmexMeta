Querying RDF¶

Use sparql.

Demonstrate creating a sparql query in Python¶
 1import os
 2from pyomexmeta import RDF, eUriType
 3
 4# get the link to teusink2000
 5teusink_biomod_link = r"https://www.ebi.ac.uk/biomodels/model/download/BIOMD0000000064.2?filename=BIOMD0000000064_url.xml"
 6
 7# download model xml, scan for rdf, create rdf graph and store in sqlite database
 8rdf = RDF.from_uri(teusink_biomod_link, syntax="rdfxml",
 9                   storage_type="sqlite", storage_name="TeusinkAnnotations2000",
10                   storage_options="new='yes'")
11
12# do a search for all annotations with glucode as resource: CHEBI:17234
13query_string = """
14SELECT ?x ?y
15WHERE {
16    ?x ?y <http://identifiers.org/obo.chebi/CHEBI:17234>
17}
18"""
19
20results_formats = [
21    "xml",
22    "json",
23    "table",
24    "csv",
25    "mkr",
26    "tsv",
27    "html",
28    "turtle",
29    "rdfxml",
30]
31
32for i in results_formats:
33    print("Sparql results format: {}".format(i))
34    print(rdf.query(query_string, results_syntax=i))
35    print("\n\n")
36

Demonstrate creating a sparql query in C++¶
 1#include "omexmeta/OmexMeta.h"
 2#include "vector"
 3using namespace omexmeta;
 4
 5int main() {
 6    //get the link to teusink2000
 7    std::string teusink_biomod_link = "https://www.ebi.ac.uk/biomodels/model/download/BIOMD0000000064.2?filename=BIOMD0000000064_url.xml";
 8
 9    // download model xml, scan for rdf, create rdf graph and store in sqlite database
10    RDF rdf = RDF::fromUri(teusink_biomod_link, "rdfxml");
11
12    std::string query_string = "SELECT ?x ?y\n"
13                               "WHERE {\n"
14                               "    ?x ?y <http://identifiers.org/obo.chebi/CHEBI:17234>\n"
15                               "}";
16
17    std::vector<std::string> results_formats = {
18            "xml",
19            "json",
20            "table",
21            "csv",
22            "mkr",
23            "tsv",
24            "html",
25            "turtle",
26            "rdfxml",
27    };
28
29    for (auto &f : results_formats){

Demonstrate creating a sparql query in C¶
 1#include "omexmeta/OmexMeta.h"
 2#include "omexmeta/OmexMetaCApi.h"
 3
 4using namespace omexmeta;
 5
 6int main() {
 7    //get the link to teusink2000
 8    const char* teusink_biomod_link = "https://www.ebi.ac.uk/biomodels/model/download/BIOMD0000000064.2?filename=BIOMD0000000064_url.xml";
 9
10    // download model xml, scan for rdf, create rdf graph and store in sqlite database
11    RDF* rdf = RDF_fromUri(teusink_biomod_link, "rdfxml");
12
13    const char* query_string = "SELECT ?x ?y\n"
14                               "WHERE {\n"
15                               "    ?x ?y <http://identifiers.org/obo.chebi/CHEBI:17234>\n"
16                               "}";
17
18    const char* results_formats[9] = {
19            "xml",
20            "json",
21            "table",
22            "csv",
23            "mkr",
24            "tsv",
25            "html",
26            "turtle",
27            "rdfxml",
28    };
29
30    for (int i=0; i<9; i++){
31        printf("Sparql results format: %s\n", results_formats[i]);
32        char* string = RDF_queryResultsAsString(rdf, query_string, results_formats[i]);

34        free(string);
35    }
36    RDF_delete(rdf);
