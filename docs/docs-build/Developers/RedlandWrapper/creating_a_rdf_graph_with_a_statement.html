Comparing Redland and its C++ Wrapper¶

The Redland Wrapper is a C++ layer around the underlying C libraries that libOmexMeta depends on. Its not perfect, not is it a comprehensive Redland Library wrapper, but for our intents and purposed it works.

Here is a demonstration of how to create a new Triple using the Redland API and the Redland Wrapper.

Create a triple store, an RDF model and add a triple to the model, using the librdf redland API.¶
 1#include "librdf.h"
 2
 3
 4int main(){
 5    librdf_world *world = librdf_new_world();
 6    librdf_storage *storage = librdf_new_storage(world, "memory", "SemsimMemoryStore", nullptr);
 7    librdf_model *model = librdf_new_model(world, storage, nullptr);
 8
 9    librdf_statement *statement = librdf_new_statement_from_nodes(
10            world,
11            librdf_new_node_from_uri_string(
12                    world, (const unsigned char *) "https://subject.com"),
13            librdf_new_node_from_uri_string(
14                    world,
15                    (const unsigned char *) "https://predicate.com"),
16            librdf_new_node_from_uri_string(
17                    world,
18                    (const unsigned char *) "https://resource.com")
19    );
20
21    librdf_model_add_statement(model, statement);
22    librdf_free_statement(statement);
23    librdf_free_storage(storage);
24    librdf_free_model(model);
25    librdf_free_world(world);
26    return 0;
27}
Create a triple store, an RDF model and add a triple to the model, using the Redland Wrapper.¶
 1#include "redland/RedlandAPI.h"
 2
 3
 4int main(){
 5    LibrdfStorage storage;
 6    LibrdfModel model(storage.get());
 7    LibrdfStatement statement = LibrdfStatement::fromRawNodePtrs(
 8            LibrdfNode::fromUriString("https//subject.com").get(),
 9            LibrdfNode::fromUriString("https//predicate.com").get(),
10            LibrdfNode::fromUriString("https//resource.com").get()
11    );
12    model.addStatement(statement);
13    statement.freeStatement();
14    model.freeModel();
15    storage.freeStorage();
16    return 0;
17}
